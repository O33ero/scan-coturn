260 int i;

266 /* loop until found or end marker (us and him both 0) */
267 for (i = 0; telnet->telopts[i].telopt != -1; ++i) {
268 if (telnet->telopts[i].telopt == telopt) {
269 if (us && telnet->telopts[i].us == TELNET_WILL251)
