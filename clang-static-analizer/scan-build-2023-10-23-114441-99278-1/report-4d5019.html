77static int stunclient_send(int sockfd, ioa_addr *local_addr, int *local_port, ioa_addr *remote_addr, int change_ip,
78 int change_port, int padding, int response_port) {
79 int ret = 0;
80
81 turn::StunMsgRequest req(STUN_METHOD_BINDING(0x0001));
82
83 req.constructBindingRequest();
84
85 if (response_port >= 0) {
86 turn::StunAttrResponsePort rpa;
87 rpa.setResponsePort((uint16_t)response_port);
88 try {
89 req.addAttr(rpa);
90 } catch (turn::WrongStunAttrFormatException &ex1) {
91 printf("Wrong rp attr format\n");
92 exit(-1);
93 } catch (turn::WrongStunBufferFormatException &ex2) {
94 printf("Wrong stun buffer format (1)\n");
95 exit(-1);
96 } catch (...) {
97 printf("Wrong something (1)\n");
98 exit(-1);
99 }
100 }
101 if (change_ip || change_port) {
102 turn::StunAttrChangeRequest cra;
103 cra.setChangeIp(change_ip);
104 cra.setChangePort(change_port);
105 try {
106 req.addAttr(cra);
107 } catch (turn::WrongStunAttrFormatException &ex1) {
108 printf("Wrong cr attr format\n");
109 exit(-1);
110 } catch (turn::WrongStunBufferFormatException &ex2) {
111 printf("Wrong stun buffer format (2)\n");
112 exit(-1);
113 } catch (...) {
114 printf("Wrong something (2)\n");
115 exit(-1);
116 }
117 }
118 if (padding) {
119 turn::StunAttrPadding pa;
120 pa.setPadding(1500);
121 try {
122 req.addAttr(pa);
123 } catch (turn::WrongStunAttrFormatException &ex1) {
124 printf("Wrong p attr format\n");
125 exit(-1);
126 } catch (turn::WrongStunBufferFormatException &ex2) {
127 printf("Wrong stun buffer format (3)\n");
128 exit(-1);
129 } catch (...) {
130 printf("Wrong something (3)\n");
131 exit(-1);
132 }
133 }
134
135 {
136 int len = 0;
137 int slen = get_ioa_addr_len(remote_addr);
138
139 do {
140 len = sendto(sockfd, req.getRawBuffer(), req.getSize(), 0, (struct sockaddr *)remote_addr, (socklen_t)slen);
141 } while (len < 0 && (socket_eintr() || socket_enobufs() || socket_eagain()));
142
143 if (len < 0)
144 err(-1, NULL((void*)0));
145 }
146
147 if (addr_get_from_sock(sockfd, local_addr) < 0) {
148 printf("%s: Cannot get address from local socket\n", __FUNCTION__);
149 } else {
150 *local_port = addr_get_port(local_addr);
151 }
152
153 return ret;
